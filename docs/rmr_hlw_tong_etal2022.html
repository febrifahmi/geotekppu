 12def AdjustedR1ucs(strength):
 13    """
 14    Adjusted R1 (adjusted uniaxial compressive rock mass strength incorporating the influence of ground water weakening and temperature environment on deep located excavation project). 
 15
 16    The equation of adjusted R1 proposed by Tong et.al (2022) is:
 17
 18    if strength (x) <= 250 MPa -> R1 = ((0.6343*math.log(x,10)) - 0.3627)
 19    if strength (x) > 250 -> R1 = 15
 20
 21    Parameters:
 22    -----------
 23
 24    - strength: uniaxial compressive strength test result of intact rock material/rock mass strength (in MPa)
 25
 26    Return:
 27    -------
 28
 29    val_r1_adj: rating value of r1 (adjusted) as a continuous rating
 30
 31    """
 32    val_r1_adj = 0
 33    if strength <= 250:
 34        val_r1_adj = 10**((0.6343*math.log(strength,10))-0.3627)
 35    elif strength > 250:
 36        val_r1_adj = 15
 37    return round(val_r1_adj,4)

 40def AdjustedR2(rqd):
 41    """
 42    Adjusted R2 - adjustment of rock quality designation rating.
 43
 44    Parameters:
 45    -----------
 46
 47    - rqd: RQD rating/value (0-100).
 48
 49    Return:
 50    -------
 51
 52    val_r2_adj: rating value of r2 (adjusted) as a continuous rating
 53
 54    """
 55    val_r2_adj = 0
 56    if rqd < 100:
 57        val_r2_adj = (0.1958*rqd) + 0.6484
 58    elif rqd == 100:
 59        val_r2_adj = 20
 60    else:
 61        val_r2_adj = None
 62    if val_r2_adj != None:
 63        return round(val_r2_adj,4)
 64    else:
 65        return val_r2_adj

 96def CalcR6(cat,favorability):
 97    """
 98    Adjustment rating for tunnel, foundation and slope based of favorability.
 99
100    Parameters:
101    -----------
102
103    - cat: category (tunnel, foundation, slope) (type String)
104    - favorability: favorability opstion (vfav -> very favorable; fav -> favorable; fair; unfav -> unfavorable; vunfav: very unfavorable)
105
106    Return:
107    -------
108
109    val_r6: rating value of R6
110
111    """
112    val_r6 = 0
113    if cat == "tunnel":
114        if favorability == "vfav":
115            val_r6 = 0
116        elif favorability == "fav":
117            val_r6 = -2
118        elif favorability == "fair":
119            val_r6 = -5
120        elif favorability == "unfav":
121            val_r6 = -10
122        elif favorability == "vunfav":
123            val_r6 = -12
124    elif cat == "foundation":
125        if favorability == "vfav":
126            val_r6 = 0
127        elif favorability == "fav":
128            val_r6 = -2
129        elif favorability == "fair":
130            val_r6 = -7
131        elif favorability == "unfav":
132            val_r6 = -15
133        elif favorability == "vunfav":
134            val_r6 = -25
135    elif cat == "slope":
136        if favorability == "vfav":
137            val_r6 = 0
138        elif favorability == "fav":
139            val_r6 = -5
140        elif favorability == "fair":
141            val_r6 = -25
142        elif favorability == "unfav":
143            val_r6 = -50
144        elif favorability == "vunfav":
145            val_r6 = -60
146    return val_r6

148def CalcR7(ri,per_i):
149    """
150    Geostress correction / strength-stress ratio index / in-situ stress modification index (R7) as proposed in Tong et.al (2022) (a ration to measure the risk of rock bursts).
151    Denoted by the equation:
152
153        R7 = Sum of (Ri x Percentage of (i))
154
155        Where Ri for specific rock burst grade:
156        I (no rock burst) --> Ri = 0
157        II (slight rock burst) --> Ri = -4
158        III (moderate rock burst) --> Ri = -8
159        IV (severe rock burst) --> Ri = -12
160
161    Parameters:
162    -----------
163
164    - Ri: score of Ri based on rock burst grade
165    - Per(i): percentage of different rock burst grade
166
167    Return:
168    -------
169
170    val_r7: value of R7 (sum of (ri * per_i)) from all data
171
172    """
173    val_r7 = ri * per_i
174    return val_r7

177def CalcR8(perm_co):
178    """
179    Rock Mass Permeability Index as main factor influence the water seepage in rocks material.
180
181    This value defined as:
182
183        R8 = -12 x (1 - Perm(<=10^-9m/s))
184
185    Parameters:
186    -----------
187
188    - perm_co: coefficient of permeability value and it should within the range <=10^-9 m/s. If permeability coefficient value == <=10^-9 m/s == 1, then R8 = -12 x (1-1) = 0. Otherwise, when permeability coefficient value == <=10^-9 m/s == 0, R8 is -12. The coefficient is between 0 and 1.
189
190
191    Return:
192    -------
193
194    val_r8: value of R8
195
196    """
197    val_r8 = -12 * (1 - perm_co)
198    return round(val_r8,2)

211
212    Return:
213    -------
214
215    val_r9: value of R9
216

222        val_r9 = -4
223    elif pH > 6 and pH < 10 and tds >= 50 and cl < 20:
224        val_r9 = -4
225    elif (pH <= 6 or pH >= 10) and tds < 50 and cl < 20:
226        val_r9 = -4
227    elif (pH <= 6 or pH >= 10) and tds >= 50 and cl < 20:
228        val_r9 = -8
229    elif (pH <= 6 or pH >= 10) and tds < 50 and cl >= 20:
230        val_r9 = -8
231    elif (pH <= 6 or pH >= 10) and tds >= 50 and cl >= 20:
232        val_r9 = -8

234        val_r9 = None
